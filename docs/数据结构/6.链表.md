```js
class Node {
  constructor(e, next) {
    this.e = e;
    this.next = next;
  }
}

class LinkedList {
  constructor() {
    this.dummyHead = new Node(null, null);
    this.size = 0;
  }

  // 向链表中index位置处添加元素
  add(index, e) {
    if (index < 0 || index > size) {
      throw new Error("Add failed. Illegal index");
    }

    let prev = this.dummyHead;
    for (let i = 0; i < index; i++) {
      prev = prev.next;
    }

    prev.next = new Node(e, prev.next);
    this.size++;
  }

  // 向链表头部添加元素
  addFirst(e) {
    this.add(0, e);
  }

  // 向链表尾部添加元素
  addLast(e) {
    this.add(this.size, e);
  }

  // 获取链表index位置处的元素
  get(index) {
    if (index < 0 || index > size) {
      throw new Error("Add failed. Illegal index");
    }

    let cur = this.dummyHead.next;
    for (let i = 0; i < index; i++) {
      cur = cur.next;
    }

    return cur.e;
  }

  // 获取链表头部的元素
  getFirst() {
    return this.get(0);
  }

  // 获取链表尾部的元素
  getLast() {
    return this.get(this.size - 1);
  }

  // 设置链表index位置处的元素
  set(index, e) {
    if (index < 0 || index > size) {
      throw new Error("Add failed. Illegal index");
    }
  }
}
```
