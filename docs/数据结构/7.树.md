## 二分搜索树

```js
class Node {
  constructor(element) {
    this.element = element;
    this.left = null;
    this.right = null;
  }
}

class BinarySearchTree {
  constructor() {
    this.root = null;
  }

  add(element) {
    const newNode = new Node(element);
    if (this.root == null) {
      this.root = newNode;
    } else {
      insert(this.root, newNode);
    }

    const insert = (node, newNode) => {
      if (node.element < newNode.element) {
        if (node.left === null) {
          node.left = newNode;
        } else {
          insert(node.left, newNode);
        }
      } else {
        if (node.right === null) {
          node.right = newNode;
        } else {
          insert(node.right, newNode);
        }
      }
    }
  }

  // 递归添加节点
  recursiveAdd(element) {
    this.root = add(null, element);
    const add = (node, element) => {
      if (node.element === null) {
        return new Node(element);
      }

      if (node.element > element) {
        node = add(node.left, element);
      } else {
        node = add(node.right, element);
      }

      return node;
    }
  }

  // 先序遍历
  preOrder(callback) {
    const traverse = (node, callback) {
      if (node !== null) {
        callback(node.element);
        traverse(node.left, callback);
        traverse(node.right, callback);
      }
    }

    traverse(this.root, callback);
  }

  // 中序遍历
  inOrder(callback) {
    const traverse = (node, callback) {
      if (node !== null) {
        traverse(node.left, callback);
        callback(node.element);
        traverse(node.right, callback);
      }
    }

    traverse(this.root, callback);
  }

  // 后序遍历
  postOrder(callback) {
    const traverse = (node, callback) {
      if (node !== null) {
        traverse(node.left, callback);
        traverse(node.right, callback);
        callback(node.element);
      }
    }

    traverse(this.root, callback);
  }
}

const tree = new BinarySearchTree();
const arr = [11, 7, 5, 3, 9, 8, 10, 13, 12, 14, 20, 18, 25];

arr.forEach(item => tree.add(arr));
tree.inOrder(value => console.log(value));
```
